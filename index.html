async function saveInlineContactChanges(id) { const donor = allDonors.find(item => Number(item.id) === Number(id)) || {}; const coldChecked = inlineChecked(id, 'cold_bonus_eligible'); const personalChecked = inlineValue(id, 'sale_type') === 'Personal Donation'; const saleTypeValue = personalChecked ? 'Personal Donation' : inlineValue(id, 'sale_type'); const historyGroupValue = inlineValue(id, 'history_group') || (coldChecked ? 'Cold / no sale since 2020' : ''); const lastAnySaleYearValue = inlineValue(id, 'last_any_sale_year') || (coldChecked ? '2020' : ''); const lastAutismSaleYearValue = inlineValue(id, 'last_autism_sale_year') || ''; const updateData = { card_id: inlineValue(id, 'card_id'), business_name: inlineValue(id, 'business_name'), contact_name: inlineValue(id, 'contact_name'), sale_contact_name: inlineValue(id, 'sale_contact_name'), mobile: inlineValue(id, 'mobile'), office_phone: inlineValue(id, 'office_phone'), email: inlineValue(id, 'email'), location: inlineValue(id, 'location'), suggested_ask: inlineValue(id, 'suggested_ask') || null, sale_number: inlineValue(id, 'sale_number') || null, sale_type: saleTypeValue, pcl_entered: inlineChecked(id, 'pcl_entered'), notes: inlineValue(id, 'notes'), check_details_reason: inlineValue(id, 'check_details_reason'), charity_name: inlineValue(id, 'charity_name') || DEFAULT_CHARITY_NAME, campaign_code: inlineValue(id, 'campaign_code') || DEFAULT_CAMPAIGN_CODE, campaign_area: inlineValue(id, 'campaign_area'), campaign_year: inlineValue(id, 'campaign_year') || DEFAULT_CAMPAIGN_YEAR, campaign_status: inlineValue(id, 'campaign_status') || 'Active', history_group: historyGroupValue, contactability_status: inlineValue(id, 'contactability_status'), research_strength: inlineValue(id, 'research_strength'), business_category: inlineValue(id, 'business_category'), source_photo: inlineValue(id, 'source_photo'), commission_rate: inlineValue(id, 'commission_rate') || null, verification_status: inlineValue(id, 'verification_status'), source_card_qa_status: inlineValue(id, 'source_card_qa_status'), phone_qa_status: inlineValue(id, 'phone_qa_status'), call_block_reason: inlineValue(id, 'call_block_reason'), local_school_angle: inlineValue(id, 'local_school_angle'), last_any_sale_year: lastAnySaleYearValue || null, last_autism_sale_year: lastAutismSaleYearValue || null, daily_call_list: inlineChecked(id, 'daily_call_list'), email_followup_stage: inlineValue(id, 'email_followup_stage') || null, last_email_followup_date: inlineValue(id, 'last_email_followup_date') || null, email_followup_note: inlineValue(id, 'email_followup_note'), sales_history: inlineValue(id, 'sales_history') }; // Preserve the existing PCL entered audit behaviour. addPclEnteredAuditFields( updateData, updateData.pcl_entered, donor ); // Preserve optional cold eligibility fields only where // those columns are available in the current database. if (hasOptionalDonorColumn('cold_bonus_eligible')) { updateData.cold_bonus_eligible = coldChecked; } if (hasOptionalDonorColumn('cold_card')) { updateData.cold_card = coldChecked; } // Preserve locked/entered sale amount handling. const amountValue = parseMoneyNumber( inlineValue(id, 'suggested_ask') ); if ( isClosedSale(donor) && amountValue > 0 ) { addOptionalSaleAmountFields( updateData, donor, amountValue ); } // Save the edit and return the actual row now stored // in Supabase. const { data: savedRow, error } = await client .from('donors') .update(updateData) .eq('id', id) .select() .single(); // Never pretend the save worked when Supabase returned // an error. if (error) { console.error( 'Inline save failed:', error ); alert( 'Failed to save row changes: ' + error.message ); return; } // Never show a fake success message when no database row // came back from the save. if (!savedRow) { console.error( 'Inline save verification failed:', { id, updateData } ); alert( 'The save could not be verified. ' + 'No changes have been assumed saved.' ); return; } // Replace the dashboard's live donor object with the // actual row returned from Supabase. updateLocalDonor( id, savedRow ); // Confirm only after the database save has been verified. showWarning( 'Row changes saved and verified.' ); // Redraw the main contact card so edits such as mobile, // main number, contact name and email appear immediately. renderDonors(); renderBonusStrip(); updateTopButtonCounts(); }
